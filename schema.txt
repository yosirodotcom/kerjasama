// ==========================================
// RELATIONSHIPS (Ref)
// ==========================================

// Pengajuan Kerjasama
Ref: T_pengajuan_kerjasama.inisiator > T_person.id

// Dokumen Kerjasama
Ref: T_dokumen_kerjasama.ref_pengajuan_kerjasama - T_pengajuan_kerjasama.id
Ref: T_dokumen_kerjasama.ref_mou - T_dokumen_kerjasama.id // Self-referencing (Turunan MoU ke PKS)
Ref: T_dokumen_kerjasama.ref_penandatangan > T_person.id
Ref: T_dokumen_kerjasama.ref_kegiatan > T_kegiatan.id

// Kegiatan
Ref: T_kegiatan.pic > T_person.id

// Mapping Ruang Lingkup
Ref: M_ruang_lingkup.id_dokumen_kerjasama > T_dokumen_kerjasama.id
Ref: M_ruang_lingkup.id_ruang_lingkup > T_ruang_lingkup.id

// Mapping Mitra Bekerjasama
Ref: M_mitra_bekerjasama.ref_pengajuan_kerjasama > T_pengajuan_kerjasama.id
Ref: M_mitra_bekerjasama.ref_mitra > T_mitra.id
Ref: M_mitra_bekerjasama.ref_pic_mitra > T_person.id
Ref: M_mitra_bekerjasama.ref_penandatangan_mitra > T_person.id

// Mapping Kegiatan Kerjasama
Ref: M_kegiatan_kerjasama.ref_dokumen_kerjasama > T_dokumen_kerjasama.id
Ref: M_kegiatan_kerjasama.ref_kegiatan > T_kegiatan.id

// Mapping Output Kegiatan
Ref: M_output_kegiatan.ref_kegiatan > T_kegiatan.id
Ref: M_output_kegiatan.ref_output > T_output.id

// Mapping Tim Pelaksana
Ref: M_tim_pelaksana.ref_T_person > T_person.id
// Catatan: ref_T_unit_pelaksana_kegiatan belum di-relasikan karena tabel T_unit_pelaksana_kegiatan tidak ada di JSON

// Mapping Unit Pelaksana Kerjasama
Ref: M_unit_pelaksana_kerjasama.ref_kegiatan > T_kegiatan.id
// Catatan: ref_unit_pelaksana_kegiatan belum di-relasikan karena tabel T_unit_pelaksana_kegiatan tidak ada di JSON
